4 　     
5 　    ' MsgBox "このシートの保護が解除されました。", vbOKOnly + _
6 　    '         vbInformation, "シート保護"
7 　    '     SendKeys "{ESC}"
8 　     
9 　     Dim V As Variant
10 　     Dim i As Integer
11 　     
12 　     '最後の値として最初の値を入れてあるのがミソ
13 　     V = Array("＊シート保護が解除されました。")
14 　 　
15 　     With Range("C1")
16 　         For i = 0 To UBound(V) - 1
17 　             '現在の値が見つかれば、配列の次の値を設定
18 　             If .Value = V(i) Then
19 　                 .Value = V(i + 1)
20 　                 Exit Sub
21 　             End If
22 　         Range("C1").Font.ColorIndex = 3
23 　         
24 　         Next
25 　         '見つからなければ、配列の最初の値を設定
26 　         .Value = V(0)
27 　     
28 　     End With
29 　 End Sub
